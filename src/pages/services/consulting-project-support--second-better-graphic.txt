    <script>
        const track = document.getElementById('gantt-track');
        const bars = [...document.querySelectorAll('.gantt-bar')];
        const rowTrack = track.querySelector('.gantt-row .flex-1');

        const dependentsMap = {
            1: [2, 3],
            2: [4],
            3: [4, 5],
            4: [5, 6],
            5: [6],
            6: []
        };

        let dragId = null;
        let mode = null;
        let startX = 0;
        let startState = {};

        function getState() {
            const state = {};
            bars.forEach(bar => {
                state[bar.dataset.id] = {
                    left: parseFloat(bar.style.left),
                    width: parseFloat(bar.style.width)
                };
            });
            return state;
        }

        function applyState(state) {
            for (const id in state) {
                const item = state[id];
                const bar = track.querySelector(`.gantt-bar[data-id="${id}"]`);
                if (bar) {
                    bar.style.left = item.left + '%';
                    bar.style.width = item.width + '%';
                    const progress = bar.querySelector('.bar-progress');
                    if (progress) progress.style.width = '100%';
                }

                const ms = document.getElementById(`ms-${id}`);
                if (ms) ms.style.left = (item.left + item.width) + '%';
            }
        }

        function cascade(parentId, state, visited = new Set()) {
            if (visited.has(parentId)) return;
            visited.add(parentId);

            const children = dependentsMap[parentId] || [];
            const parentEnd = state[parentId].left + state[parentId].width;
            const baseParentEnd = startState[parentId].left + startState[parentId].width;
            const parentShift = state[parentId].left - startState[parentId].left;
            const parentEndShift = parentEnd - baseParentEnd;

            children.forEach(childId => {
                const child = state[childId];
                const baseChild = startState[childId];

                if (mode === 'move') {
                    child.left = baseChild.left + parentShift;
                } else {
                    child.left = baseChild.left + parentEndShift;
                }

                if (child.left < 0) child.left = 0;
                if (child.left + child.width > 100) child.left = 100 - child.width;

                cascade(childId, state, visited);
            });
        }

        function onMouseDown(e) {
            const bar = e.target.closest('.gantt-bar');
            if (!bar) return;

            e.preventDefault();
            dragId = bar.dataset.id;
            bar.classList.add('dragging');

            const rect = bar.getBoundingClientRect();
            const isNearRightEdge = (e.clientX - rect.left) > (rect.width - 12);
            mode = isNearRightEdge ? 'resize' : 'move';

            startX = e.clientX;
            startState = getState();
        }

        function onMouseMove(e) {
            if (!dragId) return;

            const deltaX = e.clientX - startX;
            const trackWidth = rowTrack.offsetWidth;
            const deltaPercent = (deltaX / trackWidth) * 100;

            const newState = JSON.parse(JSON.stringify(startState));

            if (mode === 'move') {
                newState[dragId].left = startState[dragId].left + deltaPercent;
            } else {
                newState[dragId].width = startState[dragId].width + deltaPercent;
            }

            if (newState[dragId].left < 0) newState[dragId].left = 0;
            if (newState[dragId].width < 2) newState[dragId].width = 2;

            if (newState[dragId].left + newState[dragId].width > 100) {
                if (mode === 'move') newState[dragId].left = 100 - newState[dragId].width;
                else newState[dragId].width = 100 - newState[dragId].left;
            }

            cascade(dragId, newState);
            applyState(newState);
        }

        function onMouseUp() {
            if (dragId) {
                const bar = track.querySelector(`.gantt-bar[data-id="${dragId}"]`);
                if (bar) bar.classList.remove('dragging');
            }
            dragId = null;
            mode = null;
        }

        track.addEventListener('mousedown', onMouseDown);
        document.addEventListener('mousemove', onMouseMove);
        document.addEventListener('mouseup', onMouseUp);

        track.addEventListener('touchstart', e => {
            const touch = e.touches[0];
            onMouseDown({
                target: document.elementFromPoint(touch.clientX, touch.clientY),
                clientX: touch.clientX,
                preventDefault: () => e.preventDefault()
            });
        }, { passive: false });

        document.addEventListener('touchmove', e => {
            if (!dragId) return;
            e.preventDefault();
            const touch = e.touches[0];
            onMouseMove({ clientX: touch.clientX });
        }, { passive: false });

        document.addEventListener('touchend', onMouseUp);
    </script>
